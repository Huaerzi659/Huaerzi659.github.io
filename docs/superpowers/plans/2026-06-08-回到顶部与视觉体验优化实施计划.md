```js
import { readFile } from "node:fs/promises";

const checks = [
  {
    file: "src/components/BackToTop.astro",
    tokens: [
      "back-to-top",
      "aria-label=\"回到页面顶部\"",
      "scrollTo",
      "prefers-reduced-motion",
      "data-visible",
    ],
  },
  {
    file: "src/layouts/BaseLayout.astro",
    tokens: ["BackToTop", "<BackToTop />"],
  },
  {
    file: "src/styles/global.css",
    tokens: [
      "scroll-behavior: smooth",
      ".site-shell::after",
      "--motion-short",
    ],
  },
  {
    file: "src/components/Footer.astro",
    tokens: ["footer-panel::before", "footer-links a:focus-visible"],
  },
];

for (const check of checks) {
  const content = await readFile(check.file, "utf8");

  for (const token of check.tokens) {
    if (!content.includes(token)) {
      throw new Error(`${check.file} is missing ${token}`);
    }
  }
}

console.log("UX verification passed.");
```

### Task 2: BackToTop Component

**Files:**
- Create: `src/components/BackToTop.astro`
- Modify: `src/layouts/BaseLayout.astro`

- [ ] **Step 1: Create component**

Create `src/components/BackToTop.astro` as a self-contained component:

```astro
<button class="back-to-top" type="button" aria-label="回到页面顶部" data-visible="false">
  <svg aria-hidden="true" viewBox="0 0 24 24" focusable="false">
    <path d="M12 5 5.5 11.5l1.4 1.4L11 8.8V20h2V8.8l4.1 4.1 1.4-1.4L12 5Z" />
  </svg>
</button>

<script>
  const button = document.querySelector(".back-to-top");
  const motionQuery = window.matchMedia("(prefers-reduced-motion: reduce)");
  const threshold = 360;

  const updateVisibility = () => {
    button?.setAttribute("data-visible", String(window.scrollY > threshold));
  };

  button?.addEventListener("click", () => {
    window.scrollTo({
      top: 0,
      behavior: motionQuery.matches ? "auto" : "smooth",
    });
  });

  updateVisibility();
  window.addEventListener("scroll", updateVisibility, { passive: true });
</script>

<style>
  .back-to-top {
    position: fixed;
    right: max(18px, env(safe-area-inset-right));
    bottom: max(18px, env(safe-area-inset-bottom));
    z-index: 20;
    display: inline-grid;
    width: 46px;
    height: 46px;
    place-items: center;
    border: 1px solid rgba(0, 245, 255, 0.52);
    border-radius: var(--radius);
    background:
      linear-gradient(135deg, rgba(0, 245, 255, 0.18), rgba(255, 43, 214, 0.12)),
      rgba(5, 6, 15, 0.9);
    box-shadow: var(--cyan-glow);
    color: var(--neon-cyan);
    cursor: pointer;
    opacity: 0;
    pointer-events: none;
    transform: translateY(10px);
    transition: opacity var(--motion-short) var(--ease-out), transform var(--motion-short) var(--ease-out), border-color var(--motion-short) var(--ease-out), box-shadow var(--motion-short) var(--ease-out), color var(--motion-short) var(--ease-out);
  }

  .back-to-top[data-visible="true"] {
    opacity: 1;
    pointer-events: auto;
    transform: translateY(0);
  }

  .back-to-top:hover,
  .back-to-top:focus-visible {
    border-color: rgba(255, 43, 214, 0.72);
    box-shadow: var(--pink-glow);
    color: #ffd8f7;
  }

  .back-to-top svg {
    width: 22px;

  }

  @media (max-width: 560px) {

      right: 14px;

    }
  }
</style>
```
